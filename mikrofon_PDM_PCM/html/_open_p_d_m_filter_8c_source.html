115int32_t filter_table(uint8_t *data, uint8_t sincn, TPDMFilter_InitStruct *param)
116{
117 uint8_t c, i;
118 uint16_t data_index = 0;
119 uint32_t *coef_p = &coef[sincn][0];
120 int32_t F = 0;
121 uint8_t decimation = param->Decimation;
122 uint8_t channels = param->In_MicChannels;
123
124 for (i = 0; i < decimation; i += 8)
125 {
126 c = data[data_index];
127 F += ((c >> 7)) * coef_p[i] +
128 ((c >> 6) & 0x01) * coef_p[i + 1] +
129 ((c >> 5) & 0x01) * coef_p[i + 2] +
130 ((c >> 4) & 0x01) * coef_p[i + 3] +
131 ((c >> 3) & 0x01) * coef_p[i + 4] +
132 ((c >> 2) & 0x01) * coef_p[i + 5] +
133 ((c >> 1) & 0x01) * coef_p[i + 6] +
134 ((c) & 0x01) * coef_p[i + 7];
135 data_index += channels;
136 }
137 return F;
138}

141void convolve(uint32_t Signal[/* SignalLen */], unsigned short SignalLen,
142 uint32_t Kernel[/* KernelLen */], unsigned short KernelLen,
143 uint32_t Result[/* SignalLen + KernelLen - 1 */])
144{
145 uint16_t n;
146
147 for (n = 0; n < SignalLen + KernelLen - 1; n++)
148 {
149 unsigned short kmin, kmax, k;
150
151 Result[n] = 0;
152
153 kmin = (n >= KernelLen - 1) ? n - (KernelLen - 1) : 0;
154 kmax = (n < SignalLen - 1) ? n : SignalLen - 1;
155
156 for (k = kmin; k <= kmax; k++)
157 {
158 Result[n] += Signal[k] * Kernel[n - k];
159 }
160 }

163void Open_PDM_Filter_Init(TPDMFilter_InitStruct *Param)
164{
165 uint16_t i, j;
166 int64_t sum = 0;
167
168 uint8_t decimation = Param->Decimation;
169
170 for (i = 0; i < SINCN; i++)
171 {
172 Param->Coef[i] = 0;
173 Param->bit[i] = 0;
174 }
175 for (i = 0; i < decimation; i++)
176 {
177 sinc1[i] = 1;
178 }
179
180 Param->OldOut = Param->OldIn = Param->OldZ = 0;
181 Param->LP_ALFA = (Param->LP_HZ != 0 ? (uint16_t)(Param->LP_HZ * 256 / (Param->LP_HZ + Param->Fs / (2 * 3.14159))) : 0);
182 Param->HP_ALFA = (Param->HP_HZ != 0 ? (uint16_t)(Param->Fs * 256 / (2 * 3.14159 * Param->HP_HZ + Param->Fs)) : 0);
183
184 Param->FilterLen = decimation * SINCN;
185 sinc[0] = 0;
186 sinc[decimation * SINCN - 1] = 0;
187 convolve(sinc1, decimation, sinc1, decimation, sinc2);
188 convolve(sinc2, decimation * 2 - 1, sinc1, decimation, &sinc[1]);
189 for (j = 0; j < SINCN; j++)
190 {
191 for (i = 0; i < decimation; i++)
192 {
193 coef[j][i] = sinc[j * decimation + i];
194 sum += sinc[j * decimation + i];
195 }
196 }
197
198 sub_const = sum >> 1;
199 div_const = sub_const * Param->MaxVolume / 32768 / FILTER_GAIN;
200 div_const = (div_const == 0 ? 1 : div_const);
201
202#ifdef USE_LUT
203 /* Look-Up Table. */
204 uint16_t c, d, s;
205 for (s = 0; s < SINCN; s++)
206 {
207 uint32_t *coef_p = &coef[s][0];
208 for (c = 0; c < 256; c++)
209 for (d = 0; d < decimation / 8; d++)
210 lut[c][d][s] = ((c >> 7)) * coef_p[d * 8] +
211 ((c >> 6) & 0x01) * coef_p[d * 8 + 1] +
212 ((c >> 5) & 0x01) * coef_p[d * 8 + 2] +
213 ((c >> 4) & 0x01) * coef_p[d * 8 + 3] +
214 ((c >> 3) & 0x01) * coef_p[d * 8 + 4] +
215 ((c >> 2) & 0x01) * coef_p[d * 8 + 5] +
216 ((c >> 1) & 0x01) * coef_p[d * 8 + 6] +
217 ((c) & 0x01) * coef_p[d * 8 + 7];
218 }
219#endif
220}

222void Open_PDM_Filter_64(uint8_t *data, uint16_t *dataOut, uint16_t volume, TPDMFilter_InitStruct *Param)
223{
224 uint8_t i, data_out_index;
225 uint8_t channels = Param->In_MicChannels;
226 uint8_t data_inc = ((DECIMATION_MAX >> 4) * channels);
227 int64_t Z, Z0, Z1, Z2;
228 int64_t OldOut, OldIn, OldZ;
229
230 OldOut = Param->OldOut;
231 OldIn = Param->OldIn;
232 OldZ = Param->OldZ;
233
234#ifdef USE_LUT
235 uint8_t j = channels - 1;
236#endif
237
238 for (i = 0, data_out_index = 0; i < Param->Fs / 1000; i++, data_out_index += channels)
239 {
240#ifdef USE_LUT
241 Z0 = filter_tables_64[j](data, 0);
242 Z1 = filter_tables_64[j](data, 1);
243 Z2 = filter_tables_64[j](data, 2);
244#else
245 Z0 = filter_table(data, 0, Param);
246 Z1 = filter_table(data, 1, Param);
247 Z2 = filter_table(data, 2, Param);
248#endif
249
250 Z = Param->Coef[1] + Z2 - sub_const;
251 Param->Coef[1] = Param->Coef[0] + Z1;
252 Param->Coef[0] = Z0;
253
254 OldOut = (Param->HP_ALFA * (OldOut + Z - OldIn)) >> 8;
255 OldIn = Z;
256 OldZ = ((256 - Param->LP_ALFA) * OldZ + Param->LP_ALFA * OldOut) >> 8;
257
258 Z = OldZ * volume;
259 Z = RoundDiv(Z, div_const);
260 Z = SaturaLH(Z, -32700, 32700);
261
262 dataOut[data_out_index] = Z;
263 data += data_inc;
264 }
265
266 Param->OldOut = OldOut;
267 Param->OldIn = OldIn;
268 Param->OldZ = OldZ;
269}
