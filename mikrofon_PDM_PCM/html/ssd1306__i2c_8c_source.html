16/* Example code to talk to an SSD1306-based OLED display
17
18 The SSD1306 is an OLED/PLED driver chip, capable of driving displays up to
19 128x64 pixels.
20
21 NOTE: Ensure the device is capable of being driven at 3.3v NOT 5v. The Pico
22 GPIO (and therefore I2C) cannot be used at 5v.
23
24 You will need to use a level shifter on the I2C lines if you want to run the
25 board at 5v.
26
27 Connections on Raspberry Pi Pico board, other boards may vary.
28
29 GPIO PICO_DEFAULT_I2C_SDA_PIN (on Pico this is GP4 (pin 6)) -> SDA on display
30 board
31 GPIO PICO_DEFAULT_I2C_SCL_PIN (on Pico this is GP5 (pin 7)) -> SCL on
32 display board
33 3.3v (pin 36) -> VCC on display board
34 GND (pin 38) -> GND on display board
35*/

103void SSD1306_send_cmd(uint8_t cmd)
104{
105 // I2C write process expects a control byte followed by data
106 // this "data" can be a command or data to follow up a command
107 // Co = 1, D/C = 0 => the driver expects a command
108 uint8_t buf[2] = {0x80, cmd};
109 i2c_write_blocking(i2c_default, SSD1306_I2C_ADDR, buf, 2, false);
110}

112void SSD1306_send_cmd_list(uint8_t *buf, int num)
113{
114 for (int i = 0; i < num; i++)
115 SSD1306_send_cmd(buf[i]);
116}

118void SSD1306_send_buf(uint8_t buf[], int buflen)
119{
120 // in horizontal addressing mode, the column address pointer auto-increments
121 // and then wraps around to the next page, so we can send the entire frame
122 // buffer in one gooooooo!
123
124 // copy our frame buffer into a new buffer because we need to add the control byte
125 // to the beginning
126
127 uint8_t *temp_buf = malloc(buflen + 1);
128
129 temp_buf[0] = 0x40;
130 memcpy(temp_buf + 1, buf, buflen);
131
132 i2c_write_blocking(i2c_default, SSD1306_I2C_ADDR, temp_buf, buflen + 1, false);
133
134 free(temp_buf);
135}

137void SSD1306_init()
138{
139 // Some of these commands are not strictly necessary as the reset
140 // process defaults to some of these but they are shown here
141 // to demonstrate what the initialization sequence looks like
142 // Some configuration values are recommended by the board manufacturer
143
144 uint8_t cmds[] = {
145 SSD1306_SET_DISP, // set display off
146 /* memory mapping */
147 SSD1306_SET_MEM_MODE, // set memory address mode 0 = horizontal, 1 = vertical, 2 = page
148 0x00, // horizontal addressing mode
149 /* resolution and layout */
150 SSD1306_SET_DISP_START_LINE, // set display start line to 0
151 SSD1306_SET_SEG_REMAP | 0x01, // set segment re-map, column address 127 is mapped to SEG0
152 SSD1306_SET_MUX_RATIO, // set multiplex ratio
153 SSD1306_HEIGHT - 1, // Display height - 1
154 SSD1306_SET_COM_OUT_DIR | 0x08, // set COM (common) output scan direction. Scan from bottom up, COM[N-1] to COM0
155 SSD1306_SET_DISP_OFFSET, // set display offset
156 0x00, // no offset
157 SSD1306_SET_COM_PIN_CFG, // set COM (common) pins hardware configuration. Board specific magic number.
158 // 0x02 Works for 128x32, 0x12 Possibly works for 128x64. Other options 0x22, 0x32
159#if ((SSD1306_WIDTH == 128) && (SSD1306_HEIGHT == 32))
160 0x02,
161#elif ((SSD1306_WIDTH == 128) && (SSD1306_HEIGHT == 64))
162 0x12,
163#else
164 0x02,
165#endif
166 /* timing and driving scheme */
167 SSD1306_SET_DISP_CLK_DIV, // set display clock divide ratio
168 0x80, // div ratio of 1, standard freq
169 SSD1306_SET_PRECHARGE, // set pre-charge period
170 0xF1, // Vcc internally generated on our board
171 SSD1306_SET_VCOM_DESEL, // set VCOMH deselect level
172 0x30, // 0.83xVcc
173 /* display */
174 SSD1306_SET_CONTRAST, // set contrast control
175 0xFF,
176 SSD1306_SET_ENTIRE_ON, // set entire display on to follow RAM content
177 SSD1306_SET_NORM_DISP, // set normal (not inverted) display
178 SSD1306_SET_CHARGE_PUMP, // set charge pump
179 0x14, // Vcc internally generated on our board
180 SSD1306_SET_SCROLL | 0x00, // deactivate horizontal scrolling if set. This is necessary as memory writes will corrupt if scrolling was enabled
181 SSD1306_SET_DISP | 0x01, // turn display on
182 };
183
184 SSD1306_send_cmd_list(cmds, count_of(cmds));
185}

219static void SetPixel(uint8_t *buf, int x, int y, bool on)
220{
221 assert(x >= 0 && x < SSD1306_WIDTH && y >= 0 && y < SSD1306_HEIGHT);
222
223 // The calculation to determine the correct bit to set depends on which address
224 // mode we are in. This code assumes horizontal
225
226 // The video ram on the SSD1306 is split up in to 8 rows, one bit per pixel.
227 // Each row is 128 long by 8 pixels high, each byte vertically arranged, so byte 0 is x=0, y=0->7,
228 // byte 1 is x = 1, y=0->7 etc
229
230 // This code could be optimised, but is like this for clarity. The compiler
231 // should do a half decent job optimising it anyway.
232
233 const int BytesPerRow = SSD1306_WIDTH; // x pixels, 1bpp, but each row is 8 pixel high, so (x / 8) * 8
234
235 int byte_idx = (y / 8) * BytesPerRow + x;
236 uint8_t byte = buf[byte_idx];
237
238 if (on)
239 byte |= 1 << (y % 8);
240 else
241 byte &= ~(1 << (y % 8));
242
243 buf[byte_idx] = byte;
244}

246static void DrawLine(uint8_t *buf, int x0, int y0, int x1, int y1, bool on)
247{
248
249 int dx = abs(x1 - x0);
250 int sx = x0 < x1 ? 1 : -1;
251 int dy = -abs(y1 - y0);
252 int sy = y0 < y1 ? 1 : -1;
253 int err = dx + dy;
254 int e2;
255
256 while (true)
257 {
258 SetPixel(buf, x0, y0, on);
259 if (x0 == x1 && y0 == y1)
260 break;
261 e2 = 2 * err;
262
263 if (e2 >= dy)
264 {
265 err += dy;
266 x0 += sx;
267 }
268 if (e2 <= dx)
269 {
270 err += dx;
271 y0 += sy;
272 }
273 }
